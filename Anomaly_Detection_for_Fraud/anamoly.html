# Create a meshgrid

    np.linspace(X_test_pca_2[:,0].min()-1, X_test_pca_2[:,0].max()+1, 100),
    np.linspace(X_test_pca_2[:,1].min()-1, X_test_pca_2[:,1].max()+1, 100)
